#include <stdio.h>

#define ROW 3
#define COL 3

int main()
{
    int arr[ROW][COL];
    int i, j, avg;

    // Initialize the array with sample values
    arr[0][0] = 1; arr[0][1] = 2; arr[0][2] = 3;
    arr[1][0] = 4; arr[1][1] = 5; arr[1][2] = 6;
    arr[2][0] = 7; arr[2][1] = 8; arr[2][2] = 9;

    for (i = 0; i < ROW; i++)
    {
        for (j = 0; j < COL; j++)
        {
            avg = 0;
            int count = 0;

            // Sum the surrounding values
            if (i - 1 >= 0)
            {
                avg += arr[i - 1][j];
                count++;
            }
            if (j - 1 >= 0)
            {
                avg += arr[i][j - 1];
                count++;
            }
            if (i + 1 < ROW)
            {
                avg += arr[i + 1][j];
                count++;
            }
            if (j + 1 < COL)
            {
                avg += arr[i][j + 1];
                count++;
            }

            // Replace the current value with the average of surrounding values
            arr[i][j] = avg / count;
        }
    }

    // Print the modified array
    for (i = 0; i < ROW; i++)
    {
        for (j = 0; j < COL; j++)
        {
            printf("%d ", arr[i][j]);
        }
        printf("\n");
    }

    return 0;
}
